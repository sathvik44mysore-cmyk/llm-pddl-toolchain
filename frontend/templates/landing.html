Task: stack block A on block B, then put C on top of A
Generated PDDL
(:action stack
  :parameters (?b1 ?b2 - block)
  :precondition
    (and (holding ?b1)
         (clear ?b2))
  :effect
    (and (on ?b1 ?b2)
         (clear ?b1)
         (not (holding ?b1))
         (not (clear ?b2))
         (handempty)))
